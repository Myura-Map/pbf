MapLibre snippet

map.setTransformRequest((url, resourceType) => {
  if (resourceType !== 'Glyphs') return;
  const m = url.match(/(\d+)-(\d+)\.pbf$/);
  if (!m || m[1] !== '63488') return;
  return {
    url: `https://myura-map.github.io/pbf/khm/Hanuman-Regular-v1/${m[1]}-${m[2]}.pbf`
  };
});
